# present working directory
command => pwd

# all files
command => ls -a

# add file
command => touch <file_name>

# delete file
command => rm <file_name>

# delete a folder
command => rm -rf <folder_name>

# git current status
command => git status

# git last commit message change
command => git commit --amend

# git log in short
command => git log --oneline

# git hard reset (will remove everything before this commit)
command => git reset --hard <commit_id>

# git logs even after reset
command => git reflog

# git last changes revert : here commit_id refers to which commit id should be reverted
command => git revert <commit_id>
